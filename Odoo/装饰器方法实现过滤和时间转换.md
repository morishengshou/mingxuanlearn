## 一、推荐装饰器顺序

一定要这样写：

```python
@safe_external_json_fields(operation="write")
@external_json_datetime_utc(operation="write")
def write(self, vals):
    return super().write(vals)
```

原因是：

```text
外部 JSON
    ↓
先做 key 映射和字段过滤
    ↓
再对已经映射后的 Odoo Datetime 字段做 UTC 转换
    ↓
super().write()
```

如果顺序反了，例如外部 key 是 `createdAt`，而真实 Odoo 字段是 `external_created_at`，Datetime 装饰器会因为找不到 `createdAt` 这个字段而无法转换。

```python
def _safe_json_dumps(value, max_length=30000):
    try:
        text = json.dumps(value, ensure_ascii=False, default=str, indent=2)
    except Exception:
        text = repr(value)

    if max_length and len(text) > max_length:
        return text[:max_length] + "\n... truncated ..."

    return text


def _as_vals_list(vals_arg, operation):
    if operation == "write":
        if not isinstance(vals_arg, dict):
            raise ValidationError(
                _("External JSON write payload must be a dict, got %s.")
                % type(vals_arg).__name__
            )
        return [dict(vals_arg)]

    if operation == "create":
        if isinstance(vals_arg, dict):
            return [dict(vals_arg)]

        if isinstance(vals_arg, list):
            vals_list = []
            for vals in vals_arg:
                if not isinstance(vals, dict):
                    raise ValidationError(
                        _("External JSON create payload items must be dicts.")
                    )
                vals_list.append(dict(vals))
            return vals_list

        raise ValidationError(
            _("External JSON create payload must be a dict or list of dicts, got %s.")
            % type(vals_arg).__name__
        )

    raise ValueError("Unsupported operation: %s" % operation)


def _context_enabled(record, context_key):
    if context_key is None:
        return True
    return bool(record.env.context.get(context_key))


def _get_context_or_attr(record, context_key, attr_name, default=None):
    if context_key and context_key in record.env.context:
        return record.env.context.get(context_key)
    return getattr(record, attr_name, default)


def _should_raise_invalid(record):
    value = record.env.context.get("external_json_raise_on_invalid")

    if value is not None:
        return bool(value)

    return bool(getattr(record, "_external_json_raise_on_invalid", False))


def _mask_payload(record, value):
    keywords = getattr(record, "_external_json_mask_keywords", DEFAULT_MASK_KEYWORDS)
    keywords = tuple(str(keyword).lower() for keyword in keywords)

    def should_mask_key(key):
        key_text = str(key).lower()
        return any(keyword in key_text for keyword in keywords)

    def mask(obj):
        if isinstance(obj, dict):
            result = {}
            for key, val in obj.items():
                if should_mask_key(key):
                    result[key] = "***"
                else:
                    result[key] = mask(val)
            return result

        if isinstance(obj, list):
            return [mask(item) for item in obj]

        if isinstance(obj, tuple):
            return tuple(mask(item) for item in obj)

        return obj

    return mask(value)


def _log_external_json_issue(record, operation, report, res_ids=None, message=None):
    unknown_vals = report.get("unknown_vals") or {}
    ignored_vals = report.get("ignored_vals") or {}
    invalid_vals = report.get("invalid_vals") or {}

    if not unknown_vals and not ignored_vals and not invalid_vals:
        return

    unknown_keys = sorted(str(key) for key in unknown_vals.keys())
    ignored_keys = sorted(str(key) for key in ignored_vals.keys())
    invalid_keys = sorted(str(key) for key in invalid_vals.keys())

    if res_ids is None:
        res_ids = record.ids

    _logger.warning(
        "External JSON issue: model=%s, ids=%s, operation=%s, "
        "unknown_keys=%s, ignored_keys=%s, invalid_keys=%s",
        record._name,
        res_ids,
        operation,
        unknown_keys,
        ignored_keys,
        invalid_keys,
    )

    log_raw_payload = _get_context_or_attr(
        record,
        "external_json_log_raw_payload",
        "_external_json_log_raw_payload",
        True,
    )

    raw_payload = report.get("raw_payload")
    raw_payload = _mask_payload(record, raw_payload) if log_raw_payload else False

    log_model = getattr(record, "_external_json_log_model", "external.json.issue.log")

    try:
        record.env[log_model].sudo().create(
            {
                "model_name": record._name,
                "res_ids": ",".join(map(str, res_ids)) if res_ids else False,
                "operation": operation,
                "source": record.env.context.get("external_json_source"),
                "external_ref": record.env.context.get("external_json_external_ref"),
                "unknown_keys": ", ".join(unknown_keys),
                "ignored_keys": ", ".join(ignored_keys),
                "invalid_keys": ", ".join(invalid_keys),
                "unknown_payload": _safe_json_dumps(_mask_payload(record, unknown_vals)),
                "ignored_payload": _safe_json_dumps(_mask_payload(record, ignored_vals)),
                "invalid_payload": _safe_json_dumps(_mask_payload(record, invalid_vals)),
                "raw_payload": _safe_json_dumps(raw_payload) if raw_payload is not False else False,
                "message": message
                or "Some external JSON fields were ignored or failed to convert.",
            }
        )
    except Exception:
        _logger.exception("Failed to create external JSON issue log.")


def _get_field_map(record):
    return dict(getattr(record, "_external_json_field_map", {}) or {})


def _map_external_key(record, raw_key):
    return _get_field_map(record).get(raw_key, raw_key)


def _get_denied_fields(record):
    model_denied = set(getattr(record, "_external_json_denied_fields", set()) or set())
    return set(DEFAULT_DENIED_FIELDS) | model_denied


def _is_field_writable_by_external_json(record, field_name):
    field = record._fields.get(field_name)

    if not field:
        return False

    if field_name in _get_denied_fields(record):
        return False

    explicit_allowed = getattr(record, "_external_json_allowed_fields", None)

    if explicit_allowed is not None and field_name not in set(explicit_allowed):
        return False

    if getattr(field, "compute", None) and not getattr(field, "inverse", None):
        return False

    # 没有显式白名单时，进入自动保守模式
    if explicit_allowed is None:
        auto_types = getattr(
            record,
            "_external_json_auto_allowed_field_types",
            DEFAULT_AUTO_ALLOWED_FIELD_TYPES,
        )

        if field.type not in set(auto_types):
            return False

        if getattr(field, "readonly", False):
            return False

    return True


def _get_ignore_reason(record, field_name):
    field = record._fields.get(field_name)

    if not field:
        return "field does not exist on model"

    if field_name in _get_denied_fields(record):
        return "field is denied"

    explicit_allowed = getattr(record, "_external_json_allowed_fields", None)

    if explicit_allowed is not None and field_name not in set(explicit_allowed):
        return "field is not in _external_json_allowed_fields"

    if getattr(field, "compute", None) and not getattr(field, "inverse", None):
        return "computed field without inverse"

    return "field is not writable by external JSON policy"


def _sanitize_external_json_vals(record, vals):
    if not isinstance(vals, dict):
        raise ValidationError(
            _("External JSON payload must be a dict, got %s.")
            % type(vals).__name__
        )

    clean_vals = {}
    unknown_vals = {}
    ignored_vals = {}
    invalid_vals = {}
    field_sources = {}

    for raw_key, value in vals.items():
        if not isinstance(raw_key, str):
            unknown_vals[str(raw_key)] = value
            continue

        field_name = _map_external_key(record, raw_key)

        if _is_field_writable_by_external_json(record, field_name):
            if field_name in clean_vals:
                invalid_vals[raw_key] = {
                    "field": field_name,
                    "value": value,
                    "error": "duplicate external keys map to the same Odoo field; previous key: %s"
                    % field_sources[field_name],
                }

                if _should_raise_invalid(record):
                    raise ValidationError(
                        _("Duplicate external JSON keys map to field '%s'.")
                        % field_name
                    )

                continue

            field = record._fields[field_name]

            try:
                # 非 Date / Datetime 字段允许自定义转换器。
                # Date / Datetime 交给第二个装饰器统一处理。
                converter = getattr(
                    record,
                    "_external_json_convert_%s" % field_name,
                    None,
                )

                if callable(converter) and field.type not in {"date", "datetime"}:
                    value = converter(value)

                clean_vals[field_name] = value
                field_sources[field_name] = raw_key

            except Exception as exc:
                invalid_vals[raw_key] = {
                    "field": field_name,
                    "value": value,
                    "error": str(exc),
                }

                if _should_raise_invalid(record):
                    raise

        elif field_name in record._fields:
            ignored_vals[raw_key] = {
                "field": field_name,
                "value": value,
                "reason": _get_ignore_reason(record, field_name),
            }

        else:
            unknown_vals[raw_key] = value

    report = {
        "raw_payload": vals,
        "unknown_vals": unknown_vals,
        "ignored_vals": ignored_vals,
        "invalid_vals": invalid_vals,
    }

    return clean_vals, report


def safe_external_json_fields(
    operation,
    context_key="external_json_safe_write",
):
    """
    装饰器 1：安全过滤外部 JSON 字段。

    用法：
        @safe_external_json_fields(operation="write")
        def write(self, vals):
            return super().write(vals)
    """

    def decorator(method):
        @wraps(method)
        def wrapper(self, vals_arg, *args, **kwargs):
            if not _context_enabled(self, context_key):
                return method(self, vals_arg, *args, **kwargs)

            vals_list = _as_vals_list(vals_arg, operation)

            clean_vals_list = []
            reports = []

            for vals in vals_list:
                clean_vals, report = _sanitize_external_json_vals(self, vals)
                clean_vals_list.append(clean_vals)
                reports.append(report)

            if operation == "write":
                clean_vals = clean_vals_list[0]
                report = reports[0]

                _log_external_json_issue(
                    self,
                    operation="write",
                    report=report,
                    res_ids=self.ids,
                )

                if not clean_vals:
                    return True

                return method(self, clean_vals, *args, **kwargs)

            if operation == "create":
                for clean_vals, report in zip(clean_vals_list, reports):
                    if not clean_vals and not self.env.context.get(
                        "external_json_allow_empty_create",
                        False,
                    ):
                        _log_external_json_issue(
                            self,
                            operation="create",
                            report=report,
                            res_ids=[],
                            message="External JSON create skipped because no writable fields were found.",
                        )
                        raise ValidationError(
                            _(
                                "External JSON create payload contains no writable fields "
                                "for model %s."
                            )
                            % self._name
                        )

                records = method(self, clean_vals_list, *args, **kwargs)

                for record, report in zip(records, reports):
                    _log_external_json_issue(
                        record,
                        operation="create",
                        report=report,
                        res_ids=record.ids,
                    )

                return records

            raise ValueError("Unsupported operation: %s" % operation)

        return wrapper

    return decorator


def _is_empty_time_value(value):

def _parse_external_datetime(record, value, field_name):
    if _is_empty_time_value(value):
        return False

    if isinstance(value, datetime):
        return value

    if isinstance(value, date):
        return datetime.combine(value, time.min)

    if isinstance(value, bool):
        raise ValidationError(
            _("Unsupported boolean datetime value for field '%s'.") % field_name
        )

    if isinstance(value, (int, float)):
        epoch_unit = _get_epoch_unit(record)

        if epoch_unit not in {"seconds", "milliseconds"}:
            raise ValidationError(
                _(
                    "Numeric datetime value for field '%(field)s' is not enabled. "
                    "Set external_json_epoch_unit to 'seconds' or 'milliseconds'."
                )
                % {"field": field_name}
            )

        seconds = float(value)

        if epoch_unit == "milliseconds":
            seconds = seconds / 1000.0

        return datetime.fromtimestamp(seconds, tz=pytz.UTC)

    if isinstance(value, str):
        text = value.strip()

        try:
            return dateutil_parser.isoparse(text)
        except Exception:
            pass

        try:
            return fields.Datetime.to_datetime(text)
        except Exception as exc:
            raise ValidationError(
                _("Invalid datetime value for field '%(field)s': %(value)s")
                % {
                    "field": field_name,
                    "value": value,
                }
            ) from exc

    raise ValidationError(
        _("Unsupported datetime value type for field '%(field)s': %(type)s")
        % {
            "field": field_name,
            "type": type(value).__name__,
        }
    )


def _localize_naive_datetime(record, dt_value, field_name):
    source_tz = _get_source_tz(record)

    try:
        return source_tz.localize(dt_value, is_dst=None)

    except pytz.AmbiguousTimeError as exc:
        strategy = record.env.context.get("external_json_dst_strategy", "raise")

        if strategy == "standard":
            return source_tz.localize(dt_value, is_dst=False)

        if strategy == "dst":
            return source_tz.localize(dt_value, is_dst=True)

        raise ValidationError(
            _(
                "Ambiguous datetime value for field '%(field)s' "
                "in timezone '%(tz)s': %(value)s"
            )
            % {
                "field": field_name,
                "tz": source_tz.zone,
                "value": dt_value,
            }
        ) from exc

    except pytz.NonExistentTimeError as exc:
        raise ValidationError(
            _(
                "Non-existent datetime value for field '%(field)s' "
                "in timezone '%(tz)s': %(value)s"
            )
            % {
                "field": field_name,
                "tz": source_tz.zone,
                "value": dt_value,
            }
        ) from exc


def _datetime_to_utc_string(record, value, field_name):
    dt_value = _parse_external_datetime(record, value, field_name)

    if not dt_value:
        return False

    if dt_value.tzinfo is None or dt_value.utcoffset() is None:
        aware_dt = _localize_naive_datetime(record, dt_value, field_name)
    else:
        aware_dt = dt_value

    utc_naive_dt = aware_dt.astimezone(pytz.UTC).replace(tzinfo=None)

    return fields.Datetime.to_string(utc_naive_dt)


def _date_to_string(record, value, field_name):
    if _is_empty_time_value(value):
        return False

    if isinstance(value, datetime):
        return fields.Date.to_string(value.date())

    if isinstance(value, date):
        return fields.Date.to_string(value)

    if isinstance(value, str):
        text = value.strip()

        try:
            parsed_dt = dateutil_parser.isoparse(text)
            return fields.Date.to_string(parsed_dt.date())
        except Exception:
            pass

        try:
            parsed_date = fields.Date.to_date(text)
            if parsed_date:
                return fields.Date.to_string(parsed_date)
        except Exception as exc:
            raise ValidationError(
                _("Invalid date value for field '%(field)s': %(value)s")
                % {
                    "field": field_name,
                    "value": value,
                }
            ) from exc

    raise ValidationError(
        _("Unsupported date value type for field '%(field)s': %(type)s")
        % {
            "field": field_name,
            "type": type(value).__name__,
        }
    )


def _convert_datetime_values(record, vals):
    clean_vals = dict(vals)
    invalid_vals = {}

    for field_name in list(clean_vals.keys()):
        field = record._fields.get(field_name)

        if not field:
            continue

        try:
            if field.type == "datetime":
                clean_vals[field_name] = _datetime_to_utc_string(
                    record,
                    clean_vals[field_name],
                    field_name,
                )

            elif field.type == "date":
                clean_vals[field_name] = _date_to_string(
                    record,
                    clean_vals[field_name],
                    field_name,
                )

        except Exception as exc:
            invalid_vals[field_name] = {
                "value": clean_vals.get(field_name),
                "error": str(exc),
            }

            if _should_raise_invalid(record):
                raise

            clean_vals.pop(field_name, None)

    report = {
        "raw_payload": vals,
        "unknown_vals": {},
        "ignored_vals": {},
        "invalid_vals": invalid_vals,
    }

    return clean_vals, report


def external_json_datetime_utc(
    operation,
    context_key="external_json_safe_write",
):
    """
    装饰器 2：Date / Datetime 归一化。

    - fields.Datetime：转 UTC 后写入 Odoo server format。
    - fields.Date：只取日期，不做 UTC 偏移。
    """

    def decorator(method):
        @wraps(method)
        def wrapper(self, vals_arg, *args, **kwargs):
            if not _context_enabled(self, context_key):
                return method(self, vals_arg, *args, **kwargs)

            vals_list = _as_vals_list(vals_arg, operation)

            clean_vals_list = []
            reports = []

            for vals in vals_list:
                clean_vals, report = _convert_datetime_values(self, vals)
                clean_vals_list.append(clean_vals)
                reports.append(report)

            if operation == "write":
                clean_vals = clean_vals_list[0]
                report = reports[0]

                _log_external_json_issue(
                    self,
                    operation="write",
                    report=report,
                    res_ids=self.ids,
                )

                if not clean_vals:
                    return True

                return method(self, clean_vals, *args, **kwargs)

            if operation == "create":
                for clean_vals, report in zip(clean_vals_list, reports):
                    if not clean_vals and report.get("invalid_vals") and not self.env.context.get(
                        "external_json_allow_empty_create",
                        False,
                    ):
                        _log_external_json_issue(
                            self,
                            operation="create",
                            report=report,
                            res_ids=[],
                            message="External JSON create skipped because all writable values failed conversion.",
                        )
                        raise ValidationError(
                            _(
                                "External JSON create payload contains no valid writable fields "
                                "after datetime conversion for model %s."
                            )
                            % self._name
                        )

                records = method(self, clean_vals_list, *args, **kwargs)

                for record, report in zip(records, reports):
                    _log_external_json_issue(
                        record,
                        operation="create",
                        report=report,
                        res_ids=record.ids,
                    )

                return records

            raise ValueError("Unsupported operation: %s" % operation)

        return wrapper

    return decorator
```

# 四、业务模型使用示例

```python
# -*- coding: utf-8 -*-

from odoo import api, fields, models

from ..utils.external_json_decorators import (
    safe_external_json_fields,
    external_json_datetime_utc,
)


class ExternalOrder(models.Model):
    _name = "x.external.order"
    _description = "External Order"

    _external_json_allowed_fields = {
        "external_id",
        "name",
        "amount_total",
        "state",
        "external_created_at",
        "external_updated_at",
        "partner_id",
    }

    _external_json_field_map = {
        "id": "external_id",
        "createdAt": "external_created_at",
        "updatedAt": "external_updated_at",
        "customerName": "partner_id",
    }

    _external_json_source_tz = "Asia/Shanghai"
    _external_json_raise_on_invalid = False
    _external_json_epoch_unit = None
    _external_json_log_raw_payload = True

    external_id = fields.Char(index=True)
    name = fields.Char()
    amount_total = fields.Float()

    state = fields.Selection(
        [
            ("draft", "Draft"),
            ("paid", "Paid"),
            ("cancel", "Cancelled"),
        ],
        default="draft",
    )

    external_created_at = fields.Datetime()
    external_updated_at = fields.Datetime()

    partner_id = fields.Many2one("res.partner")

    def _external_json_convert_partner_id(self, value):
        if not value:
            return False

        partner = self.env["res.partner"].sudo().search(
            [
                ("name", "=", value),
            ],
            limit=1,
        )

        if not partner:
            partner = self.env["res.partner"].sudo().create(
                {
                    "name": value,
                }
            )

        return partner.id

    @api.model_create_multi
    @safe_external_json_fields(operation="create")
    @external_json_datetime_utc(operation="create")
    def create(self, vals_list):
        return super().create(vals_list)

    @safe_external_json_fields(operation="write")
    @external_json_datetime_utc(operation="write")
    def write(self, vals):
        return super().write(vals)

    def external_json_write(self, payload, source_tz=None, **extra_context):
        ctx = {
            "external_json_safe_write": True,
        }

        if source_tz:
            ctx["external_json_source_tz"] = source_tz

        ctx.update(extra_context)

        return self.with_context(**ctx).write(payload)

    @api.model
    def external_json_create(self, payload, source_tz=None, **extra_context):
        ctx = {
            "external_json_safe_write": True,
        }

        if source_tz:
            ctx["external_json_source_tz"] = source_tz

        ctx.update(extra_context)

        return self.with_context(**ctx).create(payload)
```

# 六、Datetime 行为说明

Odoo 18 官方文档说明，`Datetime` 字段在数据库中按 UTC 存储，日期时间字段写入时可以使用 Python `date/datetime`、服务端格式字符串、`False` 或 `None`；`Datetime` 的服务端格式是 `YYYY-MM-DD HH:MM:SS`。([odoo.com](https://www.odoo.com/documentation/18.0/developer/reference/backend/orm.html))

本装饰器的处理规则是：

| 字段类型 | 行为 |
|---|---|
| `fields.Datetime` | 自动转成 UTC server string |
| `fields.Date` | 只取日期，不做 UTC 偏移 |
| 无时区 datetime | 按 `external_json_source_tz` 解释 |
| 带时区 datetime | 按自身时区转 UTC |
| 数字时间戳 | 默认不启用，需要设置 `external_json_epoch_unit` |

# 七、装饰器方案和 mixin 方案对比

| 方案 | 优点 | 缺点 |
|---|---|---|
| mixin | 统一、可维护、所有公共方法集中 | 需要继承抽象模型 |
| 装饰器 | 轻量、显式、可局部使用 | 每个模型都要手动装饰 `create/write` |
| 装饰器 | 不强制改变模型继承结构 | 容易漏加装饰器 |
| 装饰器 | 可以只给部分模型启用 | 日志可能拆成字段过滤日志和时间转换日志两条 |

我的建议是：

- 如果你们团队模型很多，长期维护，**mixin 更稳**；
- 如果你们不想引入抽象模型，或者只想在少量模型上使用，**两个装饰器方案可行**；
- 如果使用装饰器，务必在代码规范中明确：所有外部映射模型的 `create()` / `write()` 必须按固定顺序加装饰器。

核心写法就是：

```python
@api.model_create_multi
@safe_external_json_fields(operation="create")
@external_json_datetime_utc(operation="create")
def create(self, vals_list):
    return super().create(vals_list)


@safe_external_json_fields(operation="write")
@external_json_datetime_utc(operation="write")
def write(self, vals):
    return super().write(vals)
```
